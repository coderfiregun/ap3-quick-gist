#include<bits/stdc++.h>
using namespace std;
 
void solve() {
  
  int n;
  cin>>n;
  vector<int> v(n+1);
  map<int, int> pos;

  for (int i=1; i<=n; ++i) {
    cin>>v[i];
    pos[v[i]] = i;
  }

  vector<char> x(n+1);
  for (int i=1; i<=n; ++i) {
    cin>>x[i];
  }


  // 6 5 1 2 3
  // 0 1 0 1 0
  // solution = 2
  // (6, 1), (1, 3)


  /* 
  Filter out what's not possible
  1. As we can only select in between l and r, we can never set s[1] = 1 or s[n] to 1 
  2. As we can only select i if min(p[l], p[r]) < p[i] < max(p[l], p[r])
     we can never set  s[pos[1]], s[pos[n]]
  
  problem cannot be solve if
  x[1] = 1, x[n] = 1, x[pos[1]] = 1, x[pos[n]] = 1
  */

  if (x[1] == '1' || x[n] == '1') {
    cout<<"-1\n";
    return;
  }

  for (int i=1; i<=n; ++i) {
    if ((v[i] == 1 || v[i] == n) && x[i] == '1') {
        cout<<"-1\n";
        return;
    }
  }


  /*
    problem can be solved with some operations
    case 1 : if we select indexes (1, pos[1]) and (1, pos[n])
    case 2 : if we select indexes (pos[1], n) and (pos[n], n)
    case 3 : if we selected indexes (pos[1], pos[n])    
    
    indexes : 1......pos[1]........pos[n].......n
    values  : x.......1.............n...........y

    case 1
        [2....x-1] in index range (1, pos[1])
        [x+1....n-1] in index range (1, pos[n])

        covering all range [2, n-1] except x
    
    case 2
        [2...y-1] in index range (pos[1], n)
        [y+1...n-1] in index range (pos[n], n)

        covering all range [2, n-1] except y

    case 3 (needed only if both case 1 and case 2 not exists)
        [2...n-1] in index range (pos[1], pos[n])
  */

  cout<<5<<"\n";
  cout<<1<<" "<<pos[1]<<"\n";
  cout<<1<<" "<<pos[n]<<"\n";
  cout<<pos[1]<<" "<<n<<"\n";
  cout<<pos[n]<<" "<<n<<"\n";
  cout<<min(pos[1], pos[n])<<" "<<max(pos[1], pos[n])<<"\n";
}   
 
 
signed main (void)
{

    for(int tc=1; tc<=ttc; ++tc){
         solve();
    }
    return 0;
}
